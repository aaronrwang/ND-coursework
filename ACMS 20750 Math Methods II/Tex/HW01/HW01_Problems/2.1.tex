    \item $xy'=y$, $y = 3$ when $x = 2$
    
    \textcolor{blue}{
    \begin{minipage}[t]{0.45\textwidth}
        General Solution:
        \begin{gather*}
            x\frac{dy}{dx} = y \\
            \frac{1}{y}dy = \frac{1}{x}dx \\
            \int \frac{1}{y} \, dy = \int \frac{1}{x} \, dx \\
            \ln|y| = \ln|x| + C \\
            e^{\ln|y|} = e^{\ln|x|+C} \\
            \textcolor{red}{
            y = Cx     
            }
        \end{gather*}
    \end{minipage}
    \hfill
    \begin{minipage}[t]{0.45\textwidth}
        Particular Solution:
        \begin{gather*}
            (3) = C(2) \\
            C = \frac{3}{2} \\
            \textcolor{red}{
            y = \frac{3}{2}x  
            }
        \end{gather*}
    \end{minipage}
