\item $(x + xy)y' + y = 0$

\textcolor{blue}{
    \begin{minipage}[t]{0.45\textwidth}
        General Solution:
        \begin{gather*}
        (x + xy)\frac{dy}{dx}= -y\\
        x(1 + y)\frac{dy}{dx}= -y\\
        \frac{1 + y}{y}dy= -xdx\\
        \Big(\frac{1}{y}+1\Big)dy= -\frac{1}{x}dx\\
        \int \Big(\frac{1}{y}+1\Big)dy= -\int \frac{1}{x}dx\\
        \ln|y| + y = -\ln|x|+C \\
        e^{\ln|y| + y} = e^{-\ln|x|+C} \\
        \textcolor{red}{
        ye^y=\frac{C}{x}
        }
        \end{gather*}
    \end{minipage}
    \hfill
    \begin{minipage}[t]{0.45\textwidth}
        Particular Solution:
        \begin{gather*}
        (1)e^{(1)}=\frac{C}{(1)}\\
        C=e\\
        \textcolor{red}{
        ye^y=\frac{e}{x}
        }
        \end{gather*}
    \end{minipage}
}
