\item $y' + 2xy^2 = 0$, $y=1$ when $x=2$

\textcolor{blue}{
    \begin{minipage}[t]{0.45\textwidth}
        General Solution:
        \begin{gather*}
            \frac{dy}{dx} + 2xy^2 = 0\\
            \frac{dy}{dx} =- 2xy^2 \\
            -\frac{dy}{y^2} = 2xdx \\
            \int-\frac{dy}{y^2} = \int2xdx \\
            \frac{1}{y}=x^2+C\\
            \textcolor{red}{
            y=\frac{1}{x^2+C}
            }
        \end{gather*}
    \end{minipage}
    \hfill
    \begin{minipage}[t]{0.45\textwidth}
        Particular Solution:
        \begin{gather*}
            1=\frac{1}{(2)^2+C}\\
            4+C=1\\
            C=-1\\
            \textcolor{red}{
                y=\frac{1}{x^2-3}
            }
        \end{gather*}
    \end{minipage}
}
