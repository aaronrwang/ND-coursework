\item [9.] $(D^2-4D+13)y=0$
\begin{gather*}
    (D^2-4D+13)y=0\\
    \text{Auxillary Equation: }(D^2-4D+13)y=0\\
    \implies (D-(2+3i))(D-(2-3i))=0\\
    \implies \text{root(s): } 2\pm 3i \implies \alpha=2;\:\beta=3\\
    \text{General Solution: } \textcolor{red}{y=e^{2x}(C_1\cos(3x)+C_2\sin(3x)})
\end{gather*}
