\item $xy'=y$
\begin{align*}
    y &= a_0 + a_1x + a_2x^2 + a_3x^3 + a_nx^n\\
    y' &= a_1 + 2a_2x + 3a_3x^2 + 4a_4x^3 + na_nx^{n-1} \\
    xy' &= a_1x + 2a_2x^2 + 3a_3x^3 + na_nx^n \\
    a_0 &= 0 \\
    a_1 &= a_1\\
    2a_2 &= a_2 \implies a_2 = 0\\
    3a_3 &= a_3 \implies a_3 = 0\\
    4a_4 &= a_4 \implies a_4 = 0\\
\end{align*}
\[
a_n = 
\begin{cases}
    a_1 & n = 1\\
    0 & \text{otherwise}
\end{cases}
\]
\textcolor{red}{\[
    y 
    = \sum_{n=0}^\infty a_nx^n
    = a_1x
\]}
