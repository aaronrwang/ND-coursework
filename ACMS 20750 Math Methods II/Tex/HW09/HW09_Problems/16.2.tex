\item [2.] $y''+4x^2y=0$
\[
    y''+4x^2y=y''+\frac{1-2a}{x}y'+\Big[(bcx^{c-1})^2+\frac{a^2-p^2c^2}{x^2}\Big]y
\]
\begin{flalign*}
1-2a = 0 \implies 2a = 1 \implies \answer{a = \frac{1}{2}}\\
(bcx^{c-1})^2=4x \implies b^2c^2x^{2c-2}=4x^2 \implies 2c - 2 =2 \implies \answer{c = 2}\\
(bcx^{c-1})^2=4x \implies b^2c^2=4 \implies b^2(2)^2=4 \implies \answer{b = 1}\\
x^2 = 0 \implies \frac{1}{2}^2-p^2(2)^2 = 0\implies 16p^2 = 1 \implies \answer{p = \frac{1}{4}}
\end{flalign*}
\answer{\[
   y = x^{1/2}Z_{1/4}(x^2) 
\]}
