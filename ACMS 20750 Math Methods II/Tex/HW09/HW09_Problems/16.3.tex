\item [3.] $xy''+2y'+4y=0$
\[
    \frac{1}{x}\Big(xy''+2y'+4y\Big) = y''+\frac{2y'}{x}+\frac{4y}{x}
\]
\[
    y''+\frac{2y'}{x}+\frac{4y}{x}=y''+\frac{1-2a}{x}y'+\Big[(bcx^{c-1})^2+\frac{a^2-p^2c^2}{x^2}\Big]y
\]
\begin{flalign*}
1-2a = 2 \implies 2a = -1  \implies \answer{a = -\frac{1}{2}}\\
(bcx^{c-1})^2=4x^{-1} \implies b^2c^2x^{2c-2}=4x^{-1} \implies 2c - 2 =-1 \implies \answer{c = \frac{1}{2}}\\
(bcx^{c-1})^2=4x^{-1} \implies b^2c^2=4 \implies b^2(\frac{1}{2})^2=4 \implies \answer{b = 4}\\
a^2-p^2c^2 = 0 \implies \Big(-\frac{1}{2}\Big)^2-p^2\Big(\frac{1}{2}\Big)^2 = 0\implies p^2 = 1 \implies \answer{p = 1}
\end{flalign*}
\answer{\[
   y = x^{-1/2}Z_{1}(4x^{1/2}) 
\]}
