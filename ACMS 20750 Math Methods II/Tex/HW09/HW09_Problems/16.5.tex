\item [5.] $\displaystyle y'' - \frac{1}{x}y' + \Big(4+\frac{1}{x^2}\Big)y=0$
\[
    \displaystyle y'' - \frac{1}{x}y' + 4y+\frac{1}{x^2}y=y''+\frac{1-2a}{x}y'+\Big[(bcx^{c-1})^2+\frac{a^2-p^2c^2}{x^2}\Big]y
\]
\begin{flalign*}
1-2a = -1 \implies 2a = 2 \implies \answer{a = 1}\\
(bcx^{c-1})^2=4 \implies b^2c^2x^{2c-2}=4 \implies 2c - 2 =0 \implies \answer{c = 1}\\
(bcx^{c-1})^2=4 \implies b^2c^2=4 \implies b^2(1)^2=4 \implies \answer{b = 2}\\
a^2-p^2c^2 = 1 \implies (1)^2-p^2(1)^2 = 1\implies 1 - p^2 = 1 \implies \answer{p = 0}
\end{flalign*}
\answer{\[
   y = xZ_{0}(2x) 
\]}
