\item [6.] $4xy'' + y=0$
\[
    \frac{1}{4x}\Big(4xy'' + y\Big)= y''+\frac{y}{4x}
\]
\[
    y''+\frac{y}{4x}=y''+\frac{1-2a}{x}y'+\Big[(bcx^{c-1})^2+\frac{a^2-p^2c^2}{x^2}\Big]y
\]
\begin{flalign*}
1-2a = 0 \implies 2a = 1 \implies \answer{a = \frac{1}{2}}\\
(bcx^{c-1})^2=\frac{1}{4x} \implies b^2c^2x^{2c-2}=\frac{1}{4x} \implies 2c - 2 =-1 \implies \answer{c = \frac{1}{2}}\\
(bcx^{c-1})^2=\frac{1}{4x} \implies b^2c^2=\frac{1}{4} \implies b^2(\frac{1}{2})^2=\frac{1}{4x} \implies \answer{b = 1}\\
a^2-p^2c^2 = 0 \implies \Big(\frac{1}{2}\Big)^2-p^2\Big(\frac{1}{2}\Big)^2 = 0\implies p^2 = 1 \implies \answer{p = 1}
\end{flalign*}
\answer{\[
   y = x^{1/2}Z_{1}(x^{1/2}) 
\]}
