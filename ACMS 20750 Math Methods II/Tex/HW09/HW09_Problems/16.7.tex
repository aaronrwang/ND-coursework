\item [7.] $xy'' + 3y' + x^3y=0$
\[
    \frac{1}{x}\Big(xy'' + 3y' + x^3y\Big)= y'' + \frac{3}{x}y' + x^2y
\]
\[
    y'' + \frac{3}{x}y' + x^2y=y''+\frac{1-2a}{x}y'+\Big[(bcx^{c-1})^2+\frac{a^2-p^2c^2}{x^2}\Big]y
\]
\begin{flalign*}
1-2a = 3 \implies 2a = -2 \implies \answer{a = -1}\\
(bcx^{c-1})^2=x^2\implies b^2c^2x^{2c-2}=x^2 \implies 2c - 2 =2 \implies \answer{c = 2}\\
(bcx^{c-1})^2=x^2 \implies b^2c^2=1 \implies b^2(2)^2=1 \implies \answer{b = \frac{1}{2}}\\
a^2-p^2c^2 = 0 \implies (-1)^2-p^2(2)^2 = 0\implies 4p^2 = 1 \implies \answer{p = \frac{1}{2}}
\end{flalign*}
\answer{\[
   y = x^{-1}Z_{1/2}(x^{2}/2) 
\]}
