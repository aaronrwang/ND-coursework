\item[1.] $z^3$
\begin{align*}
    f(z) 
    &= z^3 \\
    &= (x+iy)^3 \\
    &= \Big(x^2+2xiy+(iy)^2\Big)(x+iy)\\
    &= \Big(x^2+2xiy-y^2\Big)(x+iy)\\
    &= x^3+2x^2iy-xy^2 + x^2iy+2x(iy)^2-iy^3\\
    &= x^3+2x^2iy-xy^2 + x^2iy-2xy^2-iy^3\\
    &= x^3-3xy^2+3xiy^2-iy^3\\
    &= (x^3-3xy^2) + i(3xy^2-y^3)
\end{align*}
\[
\boxed{
\answer{
\begin{aligned}
u(x,y) &= x^3 - 3xy^2 \\
v(x,y) &= 3xy^2 - y^3
\end{aligned}
}
}
\]
