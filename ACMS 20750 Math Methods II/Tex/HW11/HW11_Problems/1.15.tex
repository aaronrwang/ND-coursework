\item[15.] $\overline{e^z}$
\begin{align*}
    f(z) 
    &= \overline{e^z} \\
    &= \overline{e^{x+iy}}\\
    &= \overline{e^x(\cos(y)+i\sin(y))}\\
    &= \overline{e^x\cos(y)+ie^x\sin(y)}\\
    &= e^x\cos(y)-ie^x\sin(y)
\end{align*}
\[
\boxed{
\answer{
\begin{aligned}
u(x,y) &= e^x\cos(y) \\
v(x,y) &= -e^x\sin(y)
\end{aligned}
}
}
\]
