\item[16.] $z^2-\overline{z^2}$
\begin{align*}
    f(z) 
    &= z^2-\overline{z^2} \\
    &= (x+iy)^2-\overline{(x+iy)^2}\\
    &= x^2+2xiy+(iy)^2-\overline{x^2+2xiy+(iy)^2}\\
    &= x^2-y^2+2xiy-\overline{x^2-y^2+2xiy}\\
    &= x^2-y^2+2xiy-(x^2-y^2-2xiy)\\
    &= i4xy\\
\end{align*}
\[
\boxed{
\answer{
\begin{aligned}
u(x,y) &= 0 \\
v(x,y) &= 4xy
\end{aligned}
}
}
\]
