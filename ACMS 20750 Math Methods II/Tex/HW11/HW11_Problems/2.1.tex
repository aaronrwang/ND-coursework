\item[1.] $z^3$
\begin{align*}
    u(x,y) &= x^3 - 3xy^2\\
    v(x,y) &= 3xy^2 - y^3
\end{align*}
\begin{align*}
    \pd{u}{x}  &= 3x^2 - 3y^2  &\qquad \pd{v}{y}  &= 3x^2 - 3y^2 \\
    \pd{u}{y} &= -6xy           &\qquad \pd{v}{x} &= 6xy
\end{align*}

\answer{
Since $\pd{u}{x} = \pd{v}{y}$ and $-\pd{u}{y} = \pd{v}{x}$, $f(z) = z^3$ is analytical.
}
