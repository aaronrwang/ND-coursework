\item[15.] $\overline{e^z}$
\begin{align*}
    u(x,y) &= e^x\cos(y) \\
    v(x,y) &= -e^x\sin(y)
\end{align*}
\begin{align*}
    \pd{u}{x}  &= e^x\cos(y)  
    &\qquad 
    \pd{v}{y}  &= -e^x\cos(y) \\ 
\end{align*}
\answer{
Since $\pd{u}{x} \neq \pd{v}{y}$, $f(z) = \overline{e^z}$ is NOT analytical.
}
