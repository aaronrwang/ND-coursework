\item[2.] $z$
\begin{align*}
    u(x,y) &= x\\
    v(x,y) &= y
\end{align*}
\begin{align*}
    \pd{u}{x}  &= 1  &\qquad \pd{v}{y}  &= 1 \\
    \pd{u}{y} &= 0           &\qquad \pd{v}{x} &= 0
\end{align*}
\answer{
Since $\pd{u}{x} = \pd{v}{y}$ and $-\pd{u}{y} = \pd{v}{x}$, $f(z) = z$ is analytical.
}
