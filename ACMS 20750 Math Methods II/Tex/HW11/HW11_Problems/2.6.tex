\item[6.] $e^z$
\begin{align*}
    u(x,y) &= e^x\cos(y)\\
    v(x,y) &= e^x\sin(y)
\end{align*}
\begin{align*}
    \pd{u}{x}  &= e^x\cos(y)  &\qquad \pd{v}{y}  &= e^x\cos(y)\\ 
    \pd{u}{y}  &= -e^x\sin(y)  &\qquad \pd{v}{x}  &= e^x\sin(y) 
\end{align*}
\answer{
Since $\pd{u}{x} = \pd{v}{y}$ and $-\pd{u}{y} = \pd{v}{x}$, $f(z) = e^z$ is analytical.
}
