\item[7.] $\cosh(z)$
\begin{align*}
    u(x,y) &= \cosh(x)\cos(y)\\
    v(x,y) &= \sinh(x)\sin(y)
\end{align*}
\begin{align*}
    \pd{u}{x}  &= \sinh(x)\cos(y)  &\qquad \pd{v}{y}  &= \sinh(x)\cos(y) \\
    \pd{u}{y}  &= -\cosh(x)\sin(y)  &\qquad \pd{v}{x}  &= \cosh(x)\sin(y) 
\end{align*}
\answer{
Since $\pd{u}{x} = \pd{v}{y}$ and $-\pd{u}{y} = \pd{v}{x}$, $f(z) = \cosh(z)$ is analytical.
}
